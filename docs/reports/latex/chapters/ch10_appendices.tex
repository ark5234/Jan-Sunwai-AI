POST /users/login
Content-Type: application/x-www-form-urlencoded

username=citizen_demo&password=citizen123
\end{verbatim}

\begin{verbatim}
POST /analyze
Cookie: js_access_token=<jwt>
# For API clients, compatibility header is also supported:
# Authorization: Bearer <jwt>
Content-Type: multipart/form-data

file=<image file>
language=en
